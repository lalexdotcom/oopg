**Step 2: Add CTE integration tests (TYPE-03)**

Test that CTE alias + use produces valid SQL:
```typescript
test('CTE alias and use produce valid SQL fragment', () => {
  const { sql, utils } = createSQLContext(db);
  const cte = utils.cte`SELECT 1 AS val`;

  // Default alias
  const fragment = cte.use();
  expect(fragment.sql).toContain('cte_');
  expect(fragment.sql).toContain('AS');
  expect(fragment.sql).toContain('SELECT 1 AS val');
  expect(fragment.values).toEqual([]);

  // Custom alias + materialized
  const named = utils.cte`SELECT 1 AS val`.alias('my_cte');
  const matFragment = named.use(true);
  expect(matFragment.sql).toContain('"my_cte"');
  expect(matFragment.sql).toContain('MATERIALIZED');

  // Not materialized
  const notMat = named.use(false);
  expect(notMat.sql).toContain('NOT MATERIALIZED');
});
```
